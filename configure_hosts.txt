# Instrucciones para configurar hosts en Windows
# ================================================

# OPCIÓN 1: Ejecutar PowerShell como Administrador
# 1. Presiona Win+X y selecciona "Windows PowerShell (Administrador)"
# 2. Ejecuta estos comandos:
Add-Content -Path "C:\Windows\System32\drivers\etc\hosts" -Value "`n127.0.0.1 bienestar.localhost"
Add-Content -Path "C:\Windows\System32\drivers\etc\hosts" -Value "`n127.0.0.1 mindcare.localhost"

# OPCIÓN 2: Editar manualmente
# 1. Abre el Bloc de notas como Administrador
# 2. Abre el archivo: C:\Windows\System32\drivers\etc\hosts
# 3. Agrega estas líneas al final:
127.0.0.1 bienestar.localhost
127.0.0.1 mindcare.localhost

# Después de cualquier opción, guarda el archivo y ciérralo
# Luego ejecuta en la terminal normal:
python test_endpoints.py
